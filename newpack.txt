/**
 * This file is main config of all automation as assist manager all package use to build this project more easily and organise also to make it easy to maintain 
 * The first line give instruction to export the configuration
 * The second line direct to the first instruction that need to be consider to run this project by webpack
 */
const currentTaskNew = process.env.npm_lifecycle_event
const path = require('path')
const {CleanWbpackPlugin} = require('clean-webpack-plugin')
const katbinMnCssExtraPlgin = require('mini-css-extract-plugin')
const katbinHtmlWbpackPlugin = require('html-webpack-plugin')
const katbinFse = require('fs-extra')

const katbinPstCssPlgins = [
    require('postcss-import'),
    require('postcss-mixins'),
    require('postcss-simple-vars'),
    require('postcss-nested'),
    require('postcss-hexrgba'),
    require('autoprefixer')
]

class RunAfterCompile{
    apply(compiler){
        compiler.hooks.done.tap('copy images', function(){
            katbinFse.copySync('./app/assets/images', './docs/assets/images')
        })
    }
}

let katbinNewCssConfig = {
    test: /\.css$/i,
    use: ['css-loader?url=false', {loader: 'postcss-loader', options: {plugins: katbinPstCssPlgins}}]
}

let pages = katbinFse.readdirSync('./app').filter(function(file){
    return file.endsWith('.html')
}).map(function(page){
    return new katbinHtmlWbpackPlugin({
        filename: page,
        template: `./app/${page}`
    })
})

let config = {
    entry: './app/assets/scripts/App.js',
    plugins: pages,
    module: {
        rules: [
            katbinNewCssConfig,
            {
                test: /\.js$/,
                exclude: /(node_modules)/,
                use: {
                    loader: 'babel-loader',
                    options: {
                        presets: ['@babel/preset-react', '@babel/preset-env']
                    }
                }
            }

        ]
    }
}

if (currentTaskNew == 'dev'){
    katbinNewCssConfig.use.unshif('style-loader')
    config.output= {
        filename: 'ktbin-bundled.js',
        path: path.resolve(__dirname, 'app')
    }

    config.devServer = {
        before: function(app, server){
            server._watch('./app/**/*.html')
        },
        contentBase: path.join(__dirname, 'app'),
        hot: true,
        port: 3000,
        host: '0.0.0.0'
    }
    config.mode = 'development'
}

if (currentTaskNew == 'build'){
    katbinNewCssConfig.use.unshift(katbinMnCssExtraPlgin.loader)
    katbinPstCssPlgins.push(require('cssnano'))
    config.output= {
        filename: '[name].[chunkhash].js',
        chunkFilename: '[name].[chunkhash].js',
        path: path.resolve(__dirname, 'docs')
    }

    config.mode = 'production'
    config.optimization = {
        splitChunks: {chunks: 'all'}
    }

    config.plugins.push(
        new CleanWbpackPlugin(), 
        new katbinMnCssExtraPlgin({filename: 'styles.[chunkhash].css'}),
        new RunAfterCompile()
    )
}

module.exports = config
